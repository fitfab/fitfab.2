import { Meta, Story, Canvas, Props, ArgsTable } from "@storybook/addon-docs";

## Carousel: Default

The default Carousel render at **100% width**, **height at 320px** and **16px gap**.

<Canvas>
  <Story name="Default">
    <div style={{ margin: "40px auto", width: "100%", maxWidth: "1312px" }}>
      <Carousel>
        {data.map((item, index) => (
          <div key={index}>
            <img src={item.src} alt={item.alt} width={item.width} />
          </div>
        ))}
      </Carousel>
    </div>
  </Story>
</Canvas>

## Carousel: Custom

The Carousel render with **780px width**, the default **height at 320px** and ** gap to 0**. Note that the width of the images are all the same(780px).

<Canvas>
  <Story name="Custom width">
    <Carousel width="780px" gap="0px">
      {fixedwidth.map((item, index) => (
        <div key={index}>
          <img src={item.src} alt={item.alt} width={item.width} />
        </div>
      ))}
    </Carousel>
  </Story>
</Canvas>
